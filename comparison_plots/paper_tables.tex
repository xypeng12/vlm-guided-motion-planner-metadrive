% ============================================================
% Smoothness Score Definition
% ============================================================
% The smoothness metric captures ride jerkiness by combining
% steering rate and longitudinal jerk.
% Lower values indicate smoother driving.
%
% Equation (paste into your paper):
\begin{equation}
  S_{\text{smooth}} = \frac{1}{2}\,\frac{\mathrm{RMS}(\Delta\delta)}{\overline{\mathrm{RMS}(\Delta\delta)}}
  \;+\; \frac{1}{2}\,\frac{\mathrm{RMS}(\Delta a)}{\overline{\mathrm{RMS}(\Delta a)}}
  \label{eq:smoothness}
\end{equation}
% where RMS(\Delta\delta) = sqrt(mean((\delta_{i+1}-\delta_i)^2)),
% and the overline denotes the cross-run mean of that component.
% Normalizing by the cross-run mean ensures both components
% contribute equally regardless of their raw magnitude.
